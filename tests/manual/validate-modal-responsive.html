CSS Implementation Reference

/* Desktop (default) */
.collection-modal-container .awsui-modal-container {
  width: 60vw !important;
  max-width: 60vw !important;
  max-height: calc(100vh - 200px) !important;
  margin-top: 100px !important;
  margin-bottom: 100px !important;
}

/* Tablet (769px - 1024px) */
@media (min-width: 769px) and (max-width: 1024px) {
  .collection-modal-container .awsui-modal-container {
    width: 75vw !important;
    max-width: 75vw !important;
  }
}

/* Mobile (< 768px) */
@media (max-width: 768px) {
  .collection-modal-container .awsui-modal-container {
    width: 90vw !important;
    max-width: 90vw !important;
  }
}

Browser DevTools Testing

Use these commands in the browser console to validate modal dimensions:

// Get modal element
const modal = document.querySelector('.collection-modal-container .awsui-modal-container');

// Check dimensions
console.log('Width:', modal.offsetWidth, 'px');
console.log('Viewport Width:', window.innerWidth, 'px');
console.log('Percentage:', (modal.offsetWidth / window.innerWidth * 100).toFixed(1) + '%');

// Check margins
const styles = window.getComputedStyle(modal);
console.log('Margin Top:', styles.marginTop);
console.log('Margin Bottom:', styles.marginBottom);
console.log('Max Height:', styles.maxHeight);
